 69    def _undersampling(
 70        self,
 71        weighting_iterations: int,
 72        weighting_normalization_strategy: str,
 73        majority_left_threshold: float,
 74        weighting_history: bool,
 75        version: int,
 76        randomize_A: float,
 77    ) -> pd.DataFrame:
 78        # Find majority class
 79        initial_class_distribution = copy(self.hg.vertices_weights)
 80        distribution = initial_class_distribution.sum(axis=0)
 81        size_diff_between_classes = int(np.abs(np.diff(distribution)[0, 0]))
 82        minority_class = distribution.argmin()
 83        if minority_class == 0:
 84            majority_class = 1
 85        elif minority_class == 1:
 86            majority_class = 0
 87        else:
 88            Warning("Multiclass classification problems are not supported.")
 89        # Calculates class-dependent vertices weights
 90        self.hg.calculate_weights(
 91            iterations=weighting_iterations,
 92            normalization_strategy=weighting_normalization_strategy,
 93            iter_history=True,
 94        )
 95        # Reduce vertex weight to single value
 96        if version == 1:
 97            if weighting_history:
 98                subtracted_weights = (
 99                    self.hg.weighting_iteration_history[0][1][
100                        :, majority_class
101                    ].todense()
102                    - self.hg.weighting_iteration_history[0][1][
103                        :, minority_class
104                    ].todense()
105                )
106                for _, vertices_weights in self.hg.weighting_iteration_history[1:]:
107                    subtracted_weights += (
108                        vertices_weights[:, majority_class].todense()
109                        - vertices_weights[:, minority_class].todense()
110                    )
111            else:
112                subtracted_weights = (
113                    self.hg.vertices_weights[:, majority_class].todense()
114                    - self.hg.vertices_weights[:, minority_class].todense()
115                )
116        elif version == 2:
117            if weighting_history:
118                subtracted_weights = (
119                    self.hg.weighting_iteration_history[0][1][
120                        :, majority_class
121                    ].todense()
122                    + self.hg.weighting_iteration_history[0][1][
123                        :, minority_class
124                    ].todense()
125                )
126                for _, vertices_weights in self.hg.weighting_iteration_history[1:]:
127                    subtracted_weights += (
128                        vertices_weights[:, majority_class].todense()
129                        + vertices_weights[:, minority_class].todense()
130                    )
131            else:
132                subtracted_weights = (
133                    self.hg.vertices_weights[:, majority_class].todense()
134                    + self.hg.vertices_weights[:, minority_class].todense()
135                )
136        elif version == 3:
137            if weighting_history:
138                subtracted_weights = self.hg.weighting_iteration_history[0][1][
139                    :, minority_class
140                ].todense()
141                for _, vertices_weights in self.hg.weighting_iteration_history[1:]:
142                    subtracted_weights += vertices_weights[:, minority_class].todense()
143            else:
144                subtracted_weights = self.hg.vertices_weights[
145                    :, minority_class
146                ].todense()
147        elif version == 4:
148            if weighting_history:
149                subtracted_weights = (
150                    self.hg.weighting_iteration_history[0][1][
151                        :, minority_class
152                    ].todense()
153                    - self.hg.weighting_iteration_history[0][1][
154                        :, majority_class
155                    ].todense()
156                )
157                for _, vertices_weights in self.hg.weighting_iteration_history[1:]:
158                    subtracted_weights += (
159                        vertices_weights[:, minority_class].todense()
160                        - vertices_weights[:, majority_class].todense()
161                    )
162            else:
163                subtracted_weights = (
164                    self.hg.vertices_weights[:, minority_class].todense()
165                    - self.hg.vertices_weights[:, majority_class].todense()
166                )
167        elif version == 5:
168            if weighting_history:
169                subtracted_weights = np.abs(
170                    self.hg.weighting_iteration_history[0][1][:, 0].todense()
171                    - self.hg.weighting_iteration_history[0][1][:, 1].todense()
172                )
173                for _, vertices_weights in self.hg.weighting_iteration_history[1:]:
174                    subtracted_weights += np.abs(
175                        vertices_weights[:, 0].todense()
176                        - vertices_weights[:, 1].todense()
177                    )
178            else:
179                subtracted_weights = (
180                    self.hg.vertices_weights[:, 0].todense()
181                    - self.hg.vertices_weights[:, 1].todense()
182                )
183        elif version == 6:
184            if weighting_history:
185                subtracted_weights = self.hg.weighting_iteration_history[0][1][
186                    :, majority_class
187                ].todense()
188                for _, vertices_weights in self.hg.weighting_iteration_history[1:]:
189                    subtracted_weights += vertices_weights[:, majority_class].todense()
190            else:
191                subtracted_weights = self.hg.vertices_weights[
192                    :, majority_class
193                ].todense()
194        else:
195            raise SyntaxError(f"Unknown scoring version: {version}")
196        # Add random selection
197        if randomize_A > 0.0:
198            # num_to_remove = int(np.max(distribution)/50)
199            # low_weight_indxs = np.argsort(subtracted_weights, axis=None)[:,:num_to_remove].tolist()[0]
200            temp_max_weight = np.amax(subtracted_weights)
201            seed(self.random_seed)
202            subtracted_weights = np.apply_along_axis(
203                lambda x: x + randomize_A * temp_max_weight * random(),
204                1,
205                subtracted_weights,
206            ).reshape(-1, 1)
207            # subtracted_weights[np.r_[low_weight_indxs]] = subtracted_weights.min()-1
208
209        # Replace minority class with max values (can't be removed)
210        subtracted_weights[
211            initial_class_distribution[:, minority_class].todense() == 1
212        ] = (subtracted_weights.max() + 1)
213        # Find vertices to drop
214        num_of_idxs_to_drop = int(
215            min(
216                size_diff_between_classes,
217                size_diff_between_classes
218                - np.ceil(size_diff_between_classes * majority_left_threshold),
219            )
220        )
221        indxs_to_drop = [
222            self.hg.vertices.inverse[i]
223            for i in subtracted_weights.argsort(axis=None)[
224                :, :num_of_idxs_to_drop
225            ].tolist()[0]
226        ]
227        return pd.concat(
228            [self.hg.X.drop(indxs_to_drop), self.hg.y.drop(indxs_to_drop)], axis=1
229        )

 59    def _modify_fi(self, fi: pd.DataFrame) -> pd.DataFrame:
 60        """Method aggregates feature value pairs to features only with summed importances.
 61
 62        Args:
 63            fi (pd.DataFrame): Pandas DataFrame where indexes are feature-value pairs and the only columns consists of feature importances.
 64
 65        Returns:
 66            pd.DataFrame: Pandas DataFrame witha aggregated features.
 67        """
 68        # Known bug - columns starting with the same name and underscore are aggregated together
 69        prefixes = self.hg.X.columns
 70        grouper = [next(p for p in prefixes if p in c) for c in list(fi.index.values)]
 71        return fi.groupby(grouper, axis=0).sum()

 73    def classifier(self, X: pd.DataFrame) -> np.ndarray:
 74        """Classification method based on the class-dependent hypergraph weighting.
 75
 76        Args:
 77            X (pd.DataFrame): Input data.
 78
 79        Raises:
 80            TypeError: Input data unknown type error.
 81            ValueError: Extracting feature names error.
 82
 83        Returns:
 84            np.ndarray: Array with predictions.
 85        """
 86        if type(X) == np.ndarray:
 87            X = pd.DataFrame(data=X, columns=self.hg.X.columns)
 88        elif type(X) != pd.DataFrame:
 89            raise TypeError(f"Unknown data type: {type(X)}")
 90
 91        X = X.astype(str)
 92
 93        # Calculate aggregated features
 94        feature_value_weights = pd.DataFrame(
 95            data=self.hg.edges_weights.todense().T,
 96            columns=list(range(len(self.hg.edges_labels))),
 97            index=self.hg.edges.keys(),
 98        )
 99        feature_weights = self._modify_fi(feature_value_weights)
100
101        def calculate_score(row):
102            try:
103                feature_value_pairs = (
104                    OneHotEncoder(drop=None, sparse=True, handle_unknown="error")
105                    .fit(row.values.reshape(1, -1))
106                    .get_feature_names_out(self.hg.X.columns)
107                )
108            except ValueError as ve:
109                raise ValueError(
110                    "Number of features does not match input data"
111                ).with_traceback(ve.__traceback__)
112
113            score_per_class = np.zeros((len(self.hg.edges_labels),))
114
115            for fv in feature_value_pairs:
116                try:
117                    score_per_class += (
118                        self.hg.edges_weights.getcol(self.hg.edges[fv]).todense().A1
119                    )
120                except KeyError:
121                    feature = next(p for p in self.hg.X.columns if p in fv)
122                    score_per_class += feature_weights.loc[
123                        feature_weights.index == feature
124                    ].values[0]
125
126            return score_per_class
127
128        return X.apply(calculate_score, axis=1, result_type="expand").values
